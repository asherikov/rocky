125  _next = nullptr;
126  if (_stack.size() == 0)
127  return;
128  T* current = _stack.top();
129  _stack.pop();
130  bool is_multi =
131  current->type == Geometry::Type::MultiLineString ||
132  current->type == Geometry::Type::MultiPoints ||
133  current->type == Geometry::Type::MultiPolygon;
134 
135  if (is_multi && _traverse_multi)
